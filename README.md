import time

def amman_and_rahma_engine(user_input):
    # قاعدة بيانات ذكية مبسطة للقيم
    values = {
        "غضب": "الرحمة تقتضي كظم الغيظ. تذكر قوله ﷺ: 'ليس الشديد بالصرعة'. حاول التنفس بعمق الآن.",
        "خوف": "الأمان هو حق أساسي. ابحث عن شخص تثق به، وتذكر أن الطمأنينة تبدأ من التوكل والعمل.",
        "مساعدة": "أنت تمارس الرحمة الآن! 'الله في عون العبد ما كان العبد في عون أخيه'.",
        "خطأ": "الرحمة تشمل ذاتك أيضاً. اعتذر إن لزم الأمر، وتعلم من درسك لتنشر الأمان حولك."
    }

    # تحليل المدخلات
    response = "لم أجد كلمة محددة، ولكن تذكر أن الأمان والرحمة هما لغة الإنسانية التي يفهمها الجميع."
    for word in values:
        if word in user_input:
            response = values[word]
            break
    
    return response
